1.1 EXTENSIONES Y TIPOS
========================

-- Habilitar PostGIS para geolocalización
CREATE EXTENSION IF NOT EXISTS postgis;

-- ENUM para tipos de pedido
CREATE TYPE order_type_enum AS ENUM (
  'VENTA_ESTANDAR',          -- Preventa vende a cliente externo
  'ABASTECIMIENTO_INTERNO',  -- Bodega surte a supermercados del dueño
  'ENTREGA_POR_CUENTA'       -- Distribuidora vendió, bodega solo entrega
);

1.2 TABLA: grupos_economicos
==============================
Propósito: Agrupar clientes del mismo dueño (Ej: Supermercado A + Distribuidora B = mismo grupo)

CREATE TABLE IF NOT EXISTS grupos_economicos (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  store_id UUID NOT NULL REFERENCES stores(id),
  nombre VARCHAR(200) NOT NULL,
  limite_credito_global NUMERIC(12,2) DEFAULT 0,
  notas TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

1.3 TABLA: grupos_clientes (Agrupación de Ruta)
=================================================
Propósito: Agrupar clientes por área/zona para asignar a preventas/ruteros
La etiqueta del grupo es LIBRE (no fija, la define el usuario)

CREATE TABLE IF NOT EXISTS grupos_clientes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  store_id UUID NOT NULL REFERENCES stores(id),
  nombre VARCHAR(200) NOT NULL,       -- Etiqueta libre: "Viejo - Sector A", "Centro", etc.
  descripcion TEXT,
  color VARCHAR(20),                  -- Para UI (color de badge)
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

1.7 TABLA: arqueos (Arqueo de Dinero a Ruteros)
==================================================
CREATE TABLE IF NOT EXISTS arqueos (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  store_id UUID NOT NULL REFERENCES stores(id),
  rutero_id UUID NOT NULL REFERENCES users(id),
  realizado_por UUID NOT NULL REFERENCES users(id),
  fecha DATE NOT NULL DEFAULT CURRENT_DATE,
  efectivo_declarado NUMERIC(12,2) NOT NULL DEFAULT 0,
  efectivo_contado NUMERIC(12,2) NOT NULL DEFAULT 0,
  diferencia NUMERIC(12,2) GENERATED ALWAYS AS (efectivo_contado - efectivo_declarado) STORED,
  cheques NUMERIC(12,2) DEFAULT 0,
  depositos NUMERIC(12,2) DEFAULT 0,
  notas TEXT,
  status VARCHAR(20) DEFAULT 'PENDIENTE',  -- PENDIENTE, CUADRADO, CON_DIFERENCIA
  created_at TIMESTAMP DEFAULT NOW()
);

1.8 TABLA: liquidaciones_ruta
===============================
CREATE TABLE IF NOT EXISTS liquidaciones_ruta (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  store_id UUID NOT NULL REFERENCES stores(id),
  rutero_id UUID NOT NULL REFERENCES users(id),
  fecha_ruta DATE NOT NULL,
  total_pedidos INT DEFAULT 0,
  total_entregados INT DEFAULT 0,
  total_rechazados INT DEFAULT 0,
  total_cobrado_contado NUMERIC(12,2) DEFAULT 0,
  total_cobrado_credito NUMERIC(12,2) DEFAULT 0,
  total_devoluciones NUMERIC(12,2) DEFAULT 0,
  efectivo_esperado NUMERIC(12,2) DEFAULT 0,
  efectivo_entregado NUMERIC(12,2) DEFAULT 0,
  diferencia NUMERIC(12,2) DEFAULT 0,
  arqueo_id UUID REFERENCES arqueos(id),
  status VARCHAR(20) DEFAULT 'PENDIENTE',  -- PENDIENTE, LIQUIDADO, CON_OBSERVACION
  liquidado_por UUID REFERENCES users(id),
  notas TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

1.9 TABLA: cargas_camion (Agrupación de Despacho)
==================================================
CREATE TABLE IF NOT EXISTS cargas_camion (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  store_id UUID NOT NULL REFERENCES stores(id),
  rutero_id UUID NOT NULL REFERENCES users(id),
  camion_placa VARCHAR(20),
  fecha_carga DATE NOT NULL DEFAULT CURRENT_DATE,
  fecha_salida TIMESTAMP,
  status VARCHAR(20) DEFAULT 'ALISTANDO',  -- ALISTANDO, CARGADO, EN_RUTA, LIQUIDADO
  total_pedidos INT DEFAULT 0,
  total_bultos INT DEFAULT 0,
  total_unidades_sueltas INT DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW()
);

1.10 TABLA: historial_asignacion_clientes
==========================================
Propósito: Trazabilidad de reasignaciones de clientes entre preventas

CREATE TABLE IF NOT EXISTS historial_asignacion_clientes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  client_id UUID NOT NULL REFERENCES clients(id),
  preventa_anterior_id UUID REFERENCES users(id),
  preventa_nuevo_id UUID REFERENCES users(id),
  motivo TEXT,
  realizado_por UUID REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW()
);

VALIDACIÓN POST-MIGRACIÓN
===========================
1. Verificar que PostGIS está habilitado: SELECT PostGIS_Version();
2. Verificar tablas nuevas: \dt grupos_*
3. Verificar columnas en clients: \d clients
4. Verificar columnas en orders: \d orders
5. Verificar JSONB en users: SELECT permisos FROM users LIMIT 1;
